-- Table film (фильмы)
-- ид	    название  жанр    страна    год   Режиссер    Актеры    длинна     Описание    путь к обложке  путь к файлу
-- film_id | title | genre | country | year | director | actors  |  duration	| description | cover  | file

-- Cоздаем таблицу фильмов: 
CREATE TABLE film (
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title varchar(150) NOT NULL,
    genre varchar(40) NOT NULL,
    country varchar(70) NOT NULL,
    year int CHECK ( year > 1900 ),
    director varchar(40) NOT NULL,
    actors varchar(400) NOT NULL,
    duration varchar(20) NOT NULL,
    description varchar NOT NULL,
    cover varchar NOT NULL,
    file varchar NOT NULL
)
